Документація

Встановлення

npm install --save shevchenko

Підключення

Node.js

var shevchenko = require("shevchenko");

Браузер

<‍script type="text/javascript" src="/path/to/shevchenko/dist/shevchenko.min.js"><‍/script>

Використання

var person = {
    gender: "male", // or "female"
    lastName: "Шевченко",
    firstName: "Тарас",
    middleName: "Григорович"
};

var result = shevchenko.inVocative(person);

console.log(result); // {lastName: 'Шевченку', firstName: 'Тарасе', middleName: 'Григоровичу'}
