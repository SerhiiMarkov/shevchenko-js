Використання
const anthroponym = {
    gender: 'male', // or 'female'
    firstName: 'Тарас',
    middleName: 'Григорович',
    lastName: 'Шевченко'
};

const result = shevchenko.inVocative(anthroponym);

console.log(result); // { gender: 'male', firstName: 'Тарасе', middleName: 'Григоровичу', lastName: 'Шевченку' }
