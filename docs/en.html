Usage Example Try it out
const input = {
  gender: 'masculine', // Grammatical gender: 'masculine' or 'feminine'.
  givenName: 'Тарас',
  patronymicName: 'Григорович',
  familyName: 'Шевченко'
};

const output = await shevchenko.inVocative(input);

console.log(output); // { givenName: "Тарасе", patronymicName: "Григоровичу", familyName: "Шевченку" }
API
shevchenko.inNominative(input: DeclensionInput): Promise<DeclensionOutput>; // Inflect in nominative grammatical case
shevchenko.inGenitive(input: DeclensionInput): Promise<DeclensionOutput>;   // Inflect in genitive grammatical case
shevchenko.inDative(input: DeclensionInput): Promise<DeclensionOutput>;     // Inflect in dative grammatical case
shevchenko.inAccusative(input: DeclensionInput): Promise<DeclensionOutput>; // Inflect in accusative grammatical case
shevchenko.inAblative(input: DeclensionInput): Promise<DeclensionOutput>;   // Inflect in ablative grammatical case
shevchenko.inLocative(input: DeclensionInput): Promise<DeclensionOutput>;   // Inflect in locative grammatical case
shevchenko.inVocative(input: DeclensionInput): Promise<DeclensionOutput>;   // Inflect in vocative grammatical case
